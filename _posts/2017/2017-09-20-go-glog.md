```go
package main

import (
	"flag"

	"github.com/golang/glog"
)

func main() {
	flag.Parse()

	glog.Info("This is a Info log")
	glog.Warning("This is a Warning log")
	glog.Error("This is a Error log")

	glog.Flush()
}
```

go run glog.go -log_dir="./"
