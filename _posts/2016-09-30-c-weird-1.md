```cpp
#include <stdio.h>

int main()
{
	printf("%c\n", "abcdef"[5]);
	printf("%c\n", 5["abcdef"]);

	char a[] = "abcdef";
	char * p = a;
	printf("%c\n", 5[a]);
	printf("%c\n", a[5]);
	printf("%c\n", 5[p]);
	printf("%c\n", p[5]);
	printf("%c\n", *(p+5));
	printf("%c\n", *(5+p));
	return 0;
}
```
